111 module mod_mpi_assemble
112 
113  use mpi
114 
115  implicit none
116 
117  public :: assemble_force_sync_comm
118  public :: assemble_force_async_comm_init
119  public :: assemble_force_async_comm_end

127 !***********************************************************************************************************************************************************************************
128  subroutine assemble_force_sync_comm()
129 !***********************************************************************************************************************************************************************************
130 
131  use mpi
132 
133  use mod_global_variables, only :&
134  tg_cpu_neighbor&
135  ,rg_gll_acceleration&
136  ,ig_ndof&
137  ,ig_ngll_total&
138  ,ig_myrank&
139  ,ig_mpi_buffer_sizemax&
140  ,ig_ncpu_neighbor
141 
142  implicit none
143 
144  real , dimension(IG_NDOF,ig_mpi_buffer_sizemax) :: dummyr
145  real , dimension(IG_NDOF,ig_mpi_buffer_sizemax,ig_ncpu_neighbor) :: dlacce
146 
147  integer, dimension(MPI_STATUS_SIZE) :: statut
148  integer :: igll
149  integer :: icon
150  integer :: ngll
151  integer :: ios
152  !
153  !
154  !******************************************
155  !fill buffer dlacce with gll to be send
156  !******************************************
157  do icon = 1,ig_ncpu_neighbor
158  do igll = 1,tg_cpu_neighbor(icon)%ngll
159 
160  dlacce(1,igll,icon) = rg_gll_acceleration(1,tg_cpu_neighbor(icon)%gll_send(igll))
161  dlacce(2,igll,icon) = rg_gll_acceleration(2,tg_cpu_neighbor(icon)%gll_send(igll))
162  dlacce(3,igll,icon) = rg_gll_acceleration(3,tg_cpu_neighbor(icon)%gll_send(igll))
163 
164  enddo
165  enddo
166  !
167  !
168  !**********************************************************
169  !synchrone comm: exchange buffers between cpu in contact
170  !**********************************************************
171  do icon = 1,ig_ncpu_neighbor
172 
173  ngll = tg_cpu_neighbor(icon)%ngll
174 
175  call mpi_sendrecv(dlacce(1,1,icon),ig_ndof*ngll,mpi_real,tg_cpu_neighbor(icon)%icpu,300,dummyr(1,1),ig_ndof*ngll,mpi_real,tg_cpu_neighbor(icon)%icpu,300,mpi_comm_world,statut,ios)
176 
177  do igll = 1,ngll
178 
179  rg_gll_acceleration(1,tg_cpu_neighbor(icon)%gll_recv(igll)) = rg_gll_acceleration(1,tg_cpu_neighbor(icon)%gll_recv(igll)) + dummyr(1,igll)
180  rg_gll_acceleration(2,tg_cpu_neighbor(icon)%gll_recv(igll)) = rg_gll_acceleration(2,tg_cpu_neighbor(icon)%gll_recv(igll)) + dummyr(2,igll)
181  rg_gll_acceleration(3,tg_cpu_neighbor(icon)%gll_recv(igll)) = rg_gll_acceleration(3,tg_cpu_neighbor(icon)%gll_recv(igll)) + dummyr(3,igll)
182 
183  enddo
184 
185  enddo
186 
187  return
188 !***********************************************************************************************************************************************************************************
189  end subroutine assemble_force_sync_comm

195 !***********************************************************************************************************************************************************************************
196  subroutine assemble_force_async_comm_init()
197 !***********************************************************************************************************************************************************************************
198 
199  use mpi
200 
201  use mod_global_variables, only :&
202  tg_cpu_neighbor&
203  ,rg_gll_acceleration&
204  ,ig_ndof&
205  ,ig_ngll_total&
206  ,ig_myrank&
207  ,ig_mpi_buffer_offset&
208  ,rg_mpi_buffer_send&
209  ,rg_mpi_buffer_recv&
210  ,ig_mpi_request_send&
211  ,ig_mpi_request_recv&
212  ,ig_mpi_buffer_sizemax&
213  ,ig_ncpu_neighbor
214 
215  implicit none
216 
217  real, dimension(IG_NDOF,ig_mpi_buffer_sizemax,ig_ncpu_neighbor) :: dlacce
218  integer :: igll
219  integer :: icon
220  integer :: ngll
221  integer :: ios
222  integer :: dof_offset
223  integer :: proc_offset
224  !
225  !
226  !*****************************************************************************************************************************************
227  !fill buffer dlacce with gll to be send
228  !*****************************************************************************************************************************************
229  do icon = 1,ig_ncpu_neighbor
230  do igll =1,tg_cpu_neighbor(icon)%ngll
231 
232  dlacce(1,igll,icon) = rg_gll_acceleration(1,tg_cpu_neighbor(icon)%gll_send(igll))
233  dlacce(2,igll,icon) = rg_gll_acceleration(2,tg_cpu_neighbor(icon)%gll_send(igll))
234  dlacce(3,igll,icon) = rg_gll_acceleration(3,tg_cpu_neighbor(icon)%gll_send(igll))
235 
236  enddo
237  enddo
238  !
239  !
240  !*****************************************************************************************************************************************
241  !asynchrone comm: exchange buffers between cpu in contact
242  !*****************************************************************************************************************************************
243  do icon = 1,ig_ncpu_neighbor
244 
245  ngll = tg_cpu_neighbor(icon)%ngll
246  proc_offset = ig_mpi_buffer_offset(icon)
247 
248  !
249  !fill buffer for cpu myrank
250  do igll = 1,ngll
251 
252  !beginning of x buffer = 0
253  dof_offset = 0
254  rg_mpi_buffer_send(proc_offset+dof_offset+igll) = dlacce(1,igll,icon)
255 
256  !beginning of y buffer = ngll
257  dof_offset = ngll
258  rg_mpi_buffer_send(proc_offset+dof_offset+igll) = dlacce(2,igll,icon)
259 
260  !beginning of z buffer = 2*ngll
261  dof_offset = 2*ngll
262  rg_mpi_buffer_send(proc_offset+dof_offset+igll) = dlacce(3,igll,icon)
263 
264  enddo
265 
266  call mpi_irecv(rg_mpi_buffer_recv(proc_offset+1),ig_ndof*ngll,mpi_real,tg_cpu_neighbor(icon)%icpu,300,mpi_comm_world,ig_mpi_request_recv(icon),ios)
267  call mpi_isend(rg_mpi_buffer_send(proc_offset+1),ig_ndof*ngll,mpi_real,tg_cpu_neighbor(icon)%icpu,300,mpi_comm_world,ig_mpi_request_send(icon),ios)
268 
269  enddo
270 
271  return
272 !***********************************************************************************************************************************************************************************
273  end subroutine assemble_force_async_comm_init

280 !***********************************************************************************************************************************************************************************
281  subroutine assemble_force_async_comm_end()
282 !***********************************************************************************************************************************************************************************
283 
284  use mpi
285 
286  use mod_global_variables, only :&
287  tg_cpu_neighbor&
288  ,rg_gll_acceleration&
289  ,ig_ndof&
290  ,ig_ngll_total&
291  ,ig_myrank&
292  ,ig_mpi_buffer_offset&
293  ,rg_mpi_buffer_send&
294  ,rg_mpi_buffer_recv&
295  ,ig_mpi_request_send&
296  ,ig_mpi_request_recv&
297  ,ig_mpi_buffer_sizemax&
298  ,ig_ncpu_neighbor
299 
300  implicit none
301 
302  integer :: igll
303  integer :: ngll
304  integer :: ios
305  integer :: dof_offset
306  integer :: proc_offset
307  integer :: nb_msg
308  integer :: r_index
309  integer, dimension(MPI_STATUS_SIZE) :: statut
310 
311  nb_msg = 0
312 
313  do while(nb_msg < ig_ncpu_neighbor)
314  !
315  !---->wait for any received message
316  call mpi_waitany(ig_ncpu_neighbor, ig_mpi_request_recv, r_index, statut, ios)
317 
318  nb_msg = nb_msg + 1
319  ngll = tg_cpu_neighbor(r_index)%ngll
320  proc_offset = ig_mpi_buffer_offset(r_index)
321  !
322  !---->sum contributions for this proc
323 
324  !beginning of x buffer = 0
325  dof_offset = 0
326  do igll = 1,ngll
327  rg_gll_acceleration(1,tg_cpu_neighbor(r_index)%gll_recv(igll)) = rg_gll_acceleration(1,tg_cpu_neighbor(r_index)%gll_recv(igll)) + rg_mpi_buffer_recv(proc_offset+dof_offset+igll)
328  enddo
329 
330  !beginning of y buffer = ngll
331  dof_offset = ngll
332  do igll = 1,ngll
333  rg_gll_acceleration(2,tg_cpu_neighbor(r_index)%gll_recv(igll)) = rg_gll_acceleration(2,tg_cpu_neighbor(r_index)%gll_recv(igll)) + rg_mpi_buffer_recv(proc_offset+dof_offset+igll)
334  enddo
335 
336  !beginning of z buffer = 2*ngll
337  dof_offset = 2*ngll
338  do igll = 1,ngll
339  rg_gll_acceleration(3,tg_cpu_neighbor(r_index)%gll_recv(igll)) = rg_gll_acceleration(3,tg_cpu_neighbor(r_index)%gll_recv(igll)) + rg_mpi_buffer_recv(proc_offset+dof_offset+igll)
340  enddo
341 
342  enddo
343 
344  return
345 !***********************************************************************************************************************************************************************************
346  end subroutine assemble_force_async_comm_end
