131 !***********************************************************************************************************************************************************************************
132  subroutine compute_double_couple_source()
133 !***********************************************************************************************************************************************************************************
134 
135  use mpi
136 
137  use mod_global_variables, only : ig_ndcsource&
138  ,ig_ngll&
139  ,ig_ndof&
140  ,tg_dcsource&
141  ,rg_dcsource_gll_force&
142  ,rg_hexa_gll_dxidx&
143  ,rg_hexa_gll_dxidy&
144  ,rg_hexa_gll_dxidz&
145  ,rg_hexa_gll_detdx&
146  ,rg_hexa_gll_detdy&
147  ,rg_hexa_gll_detdz&
148  ,rg_hexa_gll_dzedx&
149  ,rg_hexa_gll_dzedy&
150  ,rg_hexa_gll_dzedz&
151  ,ig_myrank&
152  ,error_stop&
153  ,ig_lst_unit
154 
155  use mod_init_memory
156 
157  use mod_lagrange, only : lagrad,lagrap
158 
159  implicit none
160 
161  real :: ftmp(3,3,ig_ngll,ig_ngll,ig_ngll)
162 
163  integer :: iso
164  integer :: iel
165  integer :: k
166  integer :: l
167  integer :: m
168  integer :: n
169  integer :: o
170  integer :: p
171  integer :: ios
172 
173 
174 
175  !**************************************************************************************************************!
176  !References: Komatitsch and Tromp, 1999 \cite Komatitsch1999a; Komatitsch and Tromp, 2002 \cite Komatitsch2002b!
177  !**************************************************************************************************************!
178 
179 
180  !
181  !
182  !***********************************************************
183  !implementation of the double couple point source
184  !***********************************************************
185  if (ig_myrank == 0) then
186  write(ig_lst_unit,'(" ",/,a)') "computing double-couple sources if any..."
187  call flush(ig_lst_unit)
188  endif
189 
190  if (ig_ndcsource > 0) then
191 
192  ios = init_array_real(rg_dcsource_gll_force,ig_ndcsource,ig_ngll,ig_ngll,ig_ngll,ig_ndof,"rg_dcsource_gll_force")
193 
194  endif
195 
196  !
197  !inject double couple points sources at any location (i.e., not necessarily at a gll node)
198  do iso = 1,ig_ndcsource
199 
200  iel = tg_dcsource(iso)%iel
201  !
202  !->compute first part at gll node first, then an interpolation (cf. below) is done at the xi, eta, zeta of the source
203  do k = 1,ig_ngll
204  do l = 1,ig_ngll
205  do m = 1,ig_ngll
206  ftmp(1,1,m,l,k) = tg_dcsource(iso)%mxx*rg_hexa_gll_dxidx(m,l,k,iel) &
207  + tg_dcsource(iso)%mxy*rg_hexa_gll_dxidy(m,l,k,iel) &
208  + tg_dcsource(iso)%mxz*rg_hexa_gll_dxidz(m,l,k,iel)
209  ftmp(2,1,m,l,k) = tg_dcsource(iso)%mxx*rg_hexa_gll_detdx(m,l,k,iel) &
210  + tg_dcsource(iso)%mxy*rg_hexa_gll_detdy(m,l,k,iel) &
211  + tg_dcsource(iso)%mxz*rg_hexa_gll_detdz(m,l,k,iel)
212  ftmp(3,1,m,l,k) = tg_dcsource(iso)%mxx*rg_hexa_gll_dzedx(m,l,k,iel) &
213  + tg_dcsource(iso)%mxy*rg_hexa_gll_dzedy(m,l,k,iel) &
214  + tg_dcsource(iso)%mxz*rg_hexa_gll_dzedz(m,l,k,iel)
215 
216  ftmp(1,2,m,l,k) = tg_dcsource(iso)%mxy*rg_hexa_gll_dxidx(m,l,k,iel) &
217  + tg_dcsource(iso)%myy*rg_hexa_gll_dxidy(m,l,k,iel) &
218  + tg_dcsource(iso)%myz*rg_hexa_gll_dxidz(m,l,k,iel)
219  ftmp(2,2,m,l,k) = tg_dcsource(iso)%mxy*rg_hexa_gll_detdx(m,l,k,iel) &
220  + tg_dcsource(iso)%myy*rg_hexa_gll_detdy(m,l,k,iel) &
221  + tg_dcsource(iso)%myz*rg_hexa_gll_detdz(m,l,k,iel)
222  ftmp(3,2,m,l,k) = tg_dcsource(iso)%mxy*rg_hexa_gll_dzedx(m,l,k,iel) &
223  + tg_dcsource(iso)%myy*rg_hexa_gll_dzedy(m,l,k,iel) &
224  + tg_dcsource(iso)%myz*rg_hexa_gll_dzedz(m,l,k,iel)
225 
226  ftmp(1,3,m,l,k) = tg_dcsource(iso)%mxz*rg_hexa_gll_dxidx(m,l,k,iel) &
227  + tg_dcsource(iso)%myz*rg_hexa_gll_dxidy(m,l,k,iel) &
228  + tg_dcsource(iso)%mzz*rg_hexa_gll_dxidz(m,l,k,iel)
229  ftmp(2,3,m,l,k) = tg_dcsource(iso)%mxz*rg_hexa_gll_detdx(m,l,k,iel) &
230  + tg_dcsource(iso)%myz*rg_hexa_gll_detdy(m,l,k,iel) &
231  + tg_dcsource(iso)%mzz*rg_hexa_gll_detdz(m,l,k,iel)
232  ftmp(3,3,m,l,k) = tg_dcsource(iso)%mxz*rg_hexa_gll_dzedx(m,l,k,iel) &
233  + tg_dcsource(iso)%myz*rg_hexa_gll_dzedy(m,l,k,iel) &
234  + tg_dcsource(iso)%mzz*rg_hexa_gll_dzedz(m,l,k,iel)
235  enddo
236  enddo
237  enddo
238  !
239  !->interpolation at the xi, eta, zeta of the source + multiplication by the test function
240  do k = 1,ig_ngll
241  do l = 1,ig_ngll
242  do m = 1,ig_ngll
243  rg_dcsource_gll_force(:,m,l,k,iso) = 0.0
244  do n = 1,ig_ngll
245  do o = 1,ig_ngll
246  do p = 1,ig_ngll
247  rg_dcsource_gll_force(1,m,l,k,iso) = rg_dcsource_gll_force(1,m,l,k,iso) &
248  + lagrap(p,tg_dcsource(iso)%xi,ig_ngll) &
249  * lagrap(o,tg_dcsource(iso)%et,ig_ngll) &
250  * lagrap(n,tg_dcsource(iso)%ze,ig_ngll) &
251  *(ftmp(1,1,p,o,n) * lagrad(m,tg_dcsource(iso)%xi,ig_ngll) &
252  * lagrap(l,tg_dcsource(iso)%et,ig_ngll) &
253  * lagrap(k,tg_dcsource(iso)%ze,ig_ngll) &
254  + ftmp(2,1,p,o,n) * lagrap(m,tg_dcsource(iso)%xi,ig_ngll) &
255  * lagrad(l,tg_dcsource(iso)%et,ig_ngll) &
256  * lagrap(k,tg_dcsource(iso)%ze,ig_ngll) &
257  + ftmp(3,1,p,o,n) * lagrap(m,tg_dcsource(iso)%xi,ig_ngll) &
258  * lagrap(l,tg_dcsource(iso)%et,ig_ngll) &
259  * lagrad(k,tg_dcsource(iso)%ze,ig_ngll))
260 
261  rg_dcsource_gll_force(2,m,l,k,iso) = rg_dcsource_gll_force(2,m,l,k,iso) &
262  + lagrap(p,tg_dcsource(iso)%xi,ig_ngll) &
263  * lagrap(o,tg_dcsource(iso)%et,ig_ngll) &
264  * lagrap(n,tg_dcsource(iso)%ze,ig_ngll) &
265  *(ftmp(1,2,p,o,n) * lagrad(m,tg_dcsource(iso)%xi,ig_ngll) &
266  * lagrap(l,tg_dcsource(iso)%et,ig_ngll) &
267  * lagrap(k,tg_dcsource(iso)%ze,ig_ngll) &
268  + ftmp(2,2,p,o,n) * lagrap(m,tg_dcsource(iso)%xi,ig_ngll) &
269  * lagrad(l,tg_dcsource(iso)%et,ig_ngll) &
270  * lagrap(k,tg_dcsource(iso)%ze,ig_ngll) &
271  + ftmp(3,2,p,o,n) * lagrap(m,tg_dcsource(iso)%xi,ig_ngll) &
272  * lagrap(l,tg_dcsource(iso)%et,ig_ngll) &
273  * lagrad(k,tg_dcsource(iso)%ze,ig_ngll))
274 
275  rg_dcsource_gll_force(3,m,l,k,iso) = rg_dcsource_gll_force(3,m,l,k,iso) &
276  + lagrap(p,tg_dcsource(iso)%xi,ig_ngll) &
277  * lagrap(o,tg_dcsource(iso)%et,ig_ngll) &
278  * lagrap(n,tg_dcsource(iso)%ze,ig_ngll) &
279  *(ftmp(1,3,p,o,n) * lagrad(m,tg_dcsource(iso)%xi,ig_ngll) &
280  * lagrap(l,tg_dcsource(iso)%et,ig_ngll) &
281  * lagrap(k,tg_dcsource(iso)%ze,ig_ngll) &
282  + ftmp(2,3,p,o,n) * lagrap(m,tg_dcsource(iso)%xi,ig_ngll) &
283  * lagrad(l,tg_dcsource(iso)%et,ig_ngll) &
284  * lagrap(k,tg_dcsource(iso)%ze,ig_ngll) &
285  + ftmp(3,3,p,o,n) * lagrap(m,tg_dcsource(iso)%xi,ig_ngll) &
286  * lagrap(l,tg_dcsource(iso)%et,ig_ngll) &
287  * lagrad(k,tg_dcsource(iso)%ze,ig_ngll))
288  enddo
289  enddo
290  enddo
291  enddo
292  enddo
293  enddo
294  enddo
295 
296  if (ig_myrank == 0) then
297  write(ig_lst_unit,'(a)') "done"
298  call flush(ig_lst_unit)
299  endif
300 
301  return
302 !***********************************************************************************************************************************************************************************
303  end subroutine compute_double_couple_source

314 !***********************************************************************************************************************************************************************************
315  subroutine init_double_couple_source()
316 !***********************************************************************************************************************************************************************************
317 
318  use mpi
319 
320  use mod_global_variables, only :&
321  ig_lst_unit&
322  ,tg_dcsource&
323  ,ig_ndcsource&
324  ,ig_ncpu&
325  ,ig_myrank&
326  ,type_double_couple_source&
327  ,cg_prefix&
328  ,rg_dt&
329  ,ig_ndt&
330  ,rg_pi&
331  ,rg_dcsource_user_func
332 
333  use mod_source_function , only : interp_linear
334 
335  use mod_receiver , only : search_closest_hexa_gll
336 
337  implicit none
338 
339  real, allocatable, dimension(:) :: dum1
340  real, allocatable, dimension(:) :: dum2
341  real, allocatable, dimension(:) :: dum3
342  real , dimension(ig_ncpu) :: dummy
343  real :: st
344  real :: di
345  real :: ra
346  real :: m0
347  real :: rldmin
348  real :: maxdmin
349 
350  integer , dimension(MPI_STATUS_SIZE) :: statut
351  integer :: min_loc(1)
352  integer , dimension(ig_ncpu) :: ilrcpu
353  integer :: i
354  integer :: j
355  integer :: iso
356  integer :: ios
357  integer :: ios1
358  integer :: ilndco
359  integer :: itempo
360 
361  character(len=1) :: ctempo
362 
363  type(type_double_couple_source), allocatable :: tldoco(:)
364 
365  !
366  !
367  !********************************************************************
368  !find element containing double couple point source in cpu 'myrank'
369  !********************************************************************
370 
371  ilndco = 0
372 
373  open(unit=100,file=trim(cg_prefix)//".dcs",status='old',iostat=ios)
374 
375  if (ios == 0) then
376 
377  if (ig_myrank == 0) write(ig_lst_unit,'(a)') " "
378 
379  read(unit=100,fmt='(i10)') ig_ndcsource
380 
381  allocate(tldoco(ig_ndcsource))
382 
383  do iso = 1,ig_ndcsource
384 
385  !initialize tldoco
386  tldoco(iso)%x = 0.0
387  tldoco(iso)%y = 0.0
388  tldoco(iso)%z = 0.0
389  tldoco(iso)%mxx = 0.0
390  tldoco(iso)%myy = 0.0
391  tldoco(iso)%mzz = 0.0
392  tldoco(iso)%mxy = 0.0
393  tldoco(iso)%mxz = 0.0
394  tldoco(iso)%myz = 0.0
395  tldoco(iso)%shift_time = 0.0
396  tldoco(iso)%rise_time = 0.0
397  tldoco(iso)%xi = 0.0
398  tldoco(iso)%et = 0.0
399  tldoco(iso)%ze = 0.0
400  tldoco(iso)%dmin = 0.0
401  tldoco(iso)%str = 0.0
402  tldoco(iso)%dip = 0.0
403  tldoco(iso)%rak = 0.0
404  tldoco(iso)%mw = 0.0
405  tldoco(iso)%icur = 0
406  tldoco(iso)%cpu = 0
407  tldoco(iso)%iel = 0
408  tldoco(iso)%kgll = 0
409  tldoco(iso)%lgll = 0
410  tldoco(iso)%mgll = 0
411 
412 
413  read(unit=100,fmt=*) tldoco(iso)%x,tldoco(iso)%y,tldoco(iso)%z,tldoco(iso)%mw,tldoco(iso)%str,tldoco(iso)%dip,tldoco(iso)%rak,tldoco(iso)%shift_time,tldoco(iso)%rise_time,tldoco(iso)%icur
414 
415  st = tldoco(iso)%str*rg_pi/180.0
416  di = tldoco(iso)%dip*rg_pi/180.0
417  ra = tldoco(iso)%rak*rg_pi/180.0
418  m0 = 10**(1.5*tldoco(iso)%mw + 9.1)
419  tldoco(iso)%mxx = +m0*(sin(di)*cos(ra)*sin(2.0*st) - sin(2.0*di)*sin(ra)*(cos(st))**2) !+myy for aki
420  tldoco(iso)%mxy = +m0*(sin(di)*cos(ra)*cos(2.0*st) + 0.5*sin(2.0*di)*sin(ra)* sin(2.0*st)) !+mxy
421  tldoco(iso)%mxz = +m0*(cos(di)*cos(ra)*sin( st) - cos(2.0*di)*sin(ra)* cos(st)) !-myz
422  tldoco(iso)%myy = -m0*(sin(di)*cos(ra)*sin(2.0*st) + sin(2.0*di)*sin(ra)*(sin(st))**2) !+mxx
423  tldoco(iso)%myz = +m0*(cos(di)*cos(ra)*cos( st) + cos(2.0*di)*sin(ra)* sin(st)) !-mxz
424  tldoco(iso)%mzz = +m0*sin(2.0*di)*sin(ra) !+mzz
425 
426  if ( (iso == 1) .and. ((tldoco(iso)%icur == 9) .or. (tldoco(iso)%icur == 10)) ) then
427  itempo = 0
428  open(unit=99,file=trim(cg_prefix)//".dcf")
429  do while (.true.)
430  read(unit= 99,fmt=*,iostat=ios1) ctempo
431  if (ios1 /= 0) exit
432  itempo = itempo + 1
433  enddo
434  rewind(99)
435  allocate(rg_dcsource_user_func(ig_ndt+1),dum3(ig_ndt+1),dum1(itempo),dum2(itempo))
436  do i = 1,itempo
437  read(unit=99,fmt=*) dum1(i),dum2(i)
438  enddo
439  close(99)
440  do i = 1,ig_ndt+1
441  dum3(i) = (i-1)*rg_dt
442  enddo
443  call interp_linear(1,itempo,dum1,dum2,ig_ndt+1,dum3,rg_dcsource_user_func)
444  if (ig_myrank == 0) then
445  open(unit=90,file=trim(cg_prefix)//".dcf.interp")
446  do i = 1,ig_ndt+1
447  write(unit=90,fmt='(2(E14.7,1X))') dum3(i),rg_dcsource_user_func(i)
448  enddo
449  close(90)
450  endif
451  endif
452  !
453  !->find the closest element and gll point to the source iso
454  call search_closest_hexa_gll(tldoco(iso)%x,tldoco(iso)%y,tldoco(iso)%z &
455  ,tldoco(iso)%dmin,tldoco(iso)%kgll,tldoco(iso)%lgll,tldoco(iso)%mgll,tldoco(iso)%iel)
456  !
457  !->check which cpu has the smallest dmin. this cpu will compute this source and the other won't
458  call mpi_allgather(tldoco(iso)%dmin,1,mpi_real,dummy,1,mpi_real,mpi_comm_world,ios)
459  min_loc = minloc(dummy(1:ig_ncpu))
460  if ( (min_loc(1)-1) == ig_myrank) then
461  tldoco(iso)%cpu = ig_myrank
462  ilndco = ilndco + 1
463  else
464  tldoco(iso)%cpu = -1
465  endif
466  if (ig_myrank == 0) then
467  write(ig_lst_unit,'(a,i8,a,i0)') "double couple source ",iso," computed by cpu ",min_loc(1)-1
468  endif
469  enddo
470  close(100)
471  !
472  !->transfer array tldoco which knows all the sources to array tg_dcsource that needs only to know the sources of cpu 'ig_myrank'
473  if (ilndco > 0) then
474  allocate(tg_dcsource(ilndco))
475  j = 0
476  do iso = 1,ig_ndcsource
477  if (tldoco(iso)%cpu == ig_myrank) then
478  j = j + 1
479  tg_dcsource(j) = tldoco(iso)
480  endif
481  enddo
482  ig_ndcsource = ilndco
483  else
484  ig_ndcsource = 0
485  endif
486  deallocate(tldoco)
487  !
488  !->cpu 0 gather the number of source (ig_ndcsource) of the other cpus
