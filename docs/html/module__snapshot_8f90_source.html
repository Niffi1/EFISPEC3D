215  do ix = 1,nx*ny
216  is_rec_in_cpu(ix) = .false.
217  enddo
218  endif
219

229  do iy = 1,ny

274  jrec = 0
275 
276  do iy = 1,ny
277  do ix = 1,nx
278 
279  irec = irec + 1
280 
281  if (is_rec_in_cpu(irec)) then
282 
283  jrec = jrec + 1
284  rec_x = rg_mesh_xmin + real(ix-1)*rg_receiver_snapshot_dxdy
285  rec_y = rg_mesh_ymax - real(iy-1)*rg_receiver_snapshot_dxdy
286 
287  call search_closest_quad_gll(rec_x &
288  ,rec_y &

290  ,rec_dmin &
291  ,rec_lgll &
292  ,rec_mgll &
293  ,rec_iel )

297  tg_receiver_snapshot_quad(jrec)%dmin = rec_dmin
298  tg_receiver_snapshot_quad(jrec)%iel = rec_iel
299  tg_receiver_snapshot_quad(jrec)%lgll = rec_lgll
300  tg_receiver_snapshot_quad(jrec)%mgll = rec_mgll
301  tg_receiver_snapshot_quad(jrec)%rglo = irec
302 
303  endif
304 
305  enddo
306  enddo

309 !---->compute receivers local coordinates and initialize PGx
310  if (ig_myrank == 0) then
311  write(ig_lst_unit,'(a)') " -->computing local coordinates of receivers"
312  call flush(ig_lst_unit)
313  endif
314 
315  do irec = 1,ig_nreceiver_snapshot
316 
317  call compute_info_quad_receiver(tg_receiver_snapshot_quad(irec))

320  tg_receiver_snapshot_quad(irec)%pgd_y = 0.0
321  tg_receiver_snapshot_quad(irec)%pgd_z = 0.0
322 
323  tg_receiver_snapshot_quad(irec)%pgv_x = 0.0
324  tg_receiver_snapshot_quad(irec)%pgv_y = 0.0
325  tg_receiver_snapshot_quad(irec)%pgv_z = 0.0
326 
327  tg_receiver_snapshot_quad(irec)%pga_x = 0.0
328  tg_receiver_snapshot_quad(irec)%pga_y = 0.0
329  tg_receiver_snapshot_quad(irec)%pga_z = 0.0
330 
331  enddo
332 !
333 !---->cpu0 gathers the number of receiver of all cpus. Allocation is done over all cpus to avoid seg_fault with option -check all
334  allocate(ig_receiver_snapshot_total_number(ig_ncpu))
335  allocate(ig_receiver_snapshot_mpi_shift(ig_ncpu))
336  call mpi_gather(ig_nreceiver_snapshot,1,mpi_integer,ig_receiver_snapshot_total_number,1,mpi_integer,0,mpi_comm_world,ios)
337 
338  if (ig_myrank == 0) then
339 !
340 !------->cpu0 prepares array ig_receiver_snapshot_mpi_shift for mpi_gatherv
341  ig_receiver_snapshot_mpi_shift(1) = 0
342  do icpu = 2,ig_ncpu
343  ig_receiver_snapshot_mpi_shift(icpu) = ig_receiver_snapshot_mpi_shift(icpu-1) + ig_receiver_snapshot_total_number(icpu-1)
344  enddo
345 
346  endif
347 !
348 !---->cpu0 gathers global number of all snapshot receivers of all cpus (because cpu0 writes the *.grd files alone)
349  allocate(ig_receiver_snapshot_glonum(nx*ny),stat=ios)
350  if (ios /= 0) then
351  write(info,'(a)') "error in subroutine init_snapshot while allocating ig_receiver_snapshot_glonum"
352  call error_stop(info)
353  endif
354 !
355 !---->cpu0 gathers z-coordinate of all snapshot receivers of all cpus (because cpu0 writes the VTK files alone)
356  allocate(rldum3(nx*ny),stat=ios)
357  if (ios /= 0) then
358  write(info,'(a)') "error in subroutine init_snapshot while allocating rldum3"
359  call error_stop(info)
360  endif
361 !
362 !---->cpu0 gathers global number and z-coordinate of all snapshot receivers of all cpus
363  allocate(ildum1(ig_nreceiver_snapshot),stat=ios)
364  allocate(rldum2(ig_nreceiver_snapshot),stat=ios)
365  if (ios /= 0) then
366  write(info,'(a)') "error in subroutine init_snapshot while allocating *dum*"
367  call error_stop(info)
368  else
369  do irec = 1,ig_nreceiver_snapshot
370  ildum1(irec) = tg_receiver_snapshot_quad(irec)%rglo
371  rldum2(irec) = tg_receiver_snapshot_quad(irec)%z

373  endif
374 
375  call mpi_gatherv(ildum1 &
376  ,ig_nreceiver_snapshot &
377  ,mpi_integer &
378  ,ig_receiver_snapshot_glonum &
379  ,ig_receiver_snapshot_total_number &
380  ,ig_receiver_snapshot_mpi_shift &
381  ,mpi_integer &
382  ,0 &
383  ,mpi_comm_world &
384  ,ios)
385 
386  call mpi_gatherv(rldum2 &
387  ,ig_nreceiver_snapshot &
388  ,mpi_real &
389  ,rldum3 &
390  ,ig_receiver_snapshot_total_number &
391  ,ig_receiver_snapshot_mpi_shift &
392  ,mpi_real &
393  ,0 &
394  ,mpi_comm_world &
395  ,ios)
396 
397  deallocate(rldum2)
398 
399  if (ig_myrank == 0) then
400 
401  allocate(rg_receiver_snapshot_z(nx*ny),stat=ios)
402  if (ios /= 0) then
403  write(info,'(a)') "error in subroutine init_snapshot while allocating rg_receiver_snapshot_z"
404  call error_stop(info)
405  endif
406 
407  allocate(ig_receiver_snapshot_locnum(rg_receiver_snapshot_nx*rg_receiver_snapshot_ny),stat=ios)
408  if (ios /= 0) then
409  write(info,'(a)') "error in subroutine init_snapshot while allocating ig_receiver_snapshot_locnum"
410  call error_stop(info)
411  endif
412 
413  irec = 0
414 
415  do iy = 1,rg_receiver_snapshot_ny
416  do ix = 1,rg_receiver_snapshot_nx
417 
418  irec = irec + 1
419 
420  do iz = 1,rg_receiver_snapshot_nx*rg_receiver_snapshot_ny
421 
422  if (ig_receiver_snapshot_glonum(iz) == irec) then
423 
424  ig_receiver_snapshot_locnum(irec) = iz
425  rg_receiver_snapshot_z(irec) = rldum3(iz)
426 
427  endif
428 
429  enddo
430 
431  enddo
432  enddo
433 
434  endif
435 
436  if (ig_myrank == 0) then
437  write(ig_lst_unit,'(a)') "done"
438  call flush(ig_lst_unit)
439  endif
440 
441  return

453  subroutine write_snapshot()
454 !***********************************************************************************************************************************************************************************
455 
456  use mpi
457 
458  use mod_global_variables, only :&
459  lg_snapshot_gmt&
460  ,lg_snapshot_vtk&
461  ,ig_ndof&
462  ,ig_ngll&
463  ,ig_idt&

466  ,lg_snapshot_velocity&
467  ,lg_snapshot_acceleration&
468  ,ig_nreceiver_snapshot&

471  ,rg_gll_displacement&
472  ,rg_gll_velocity&
473  ,rg_gll_acceleration&
474  ,ig_myrank
475 
476  use mod_gll_value , only : get_quad_gll_value
477 
478  use mod_lagrange , only : quad_lagrange_interp
479 
480  implicit none
481 
482  real, dimension(IG_NDOF,IG_NGLL,IG_NGLL,IG_NGLL) :: gll_dis
483  real, dimension(IG_NDOF,IG_NGLL,IG_NGLL,IG_NGLL) :: gll_vel
484  real, dimension(IG_NDOF,IG_NGLL,IG_NGLL,IG_NGLL) :: gll_acc
485 
486  real, dimension(ig_nreceiver_snapshot) :: ux
487  real, dimension(ig_nreceiver_snapshot) :: uy
488  real, dimension(ig_nreceiver_snapshot) :: uz
489  real, dimension(ig_nreceiver_snapshot) :: vx
490  real, dimension(ig_nreceiver_snapshot) :: vy
491  real, dimension(ig_nreceiver_snapshot) :: vz
492  real, dimension(ig_nreceiver_snapshot) :: ax
493  real, dimension(ig_nreceiver_snapshot) :: ay
494  real, dimension(ig_nreceiver_snapshot) :: az
495 
496  integer :: irec
497 
498  character(len=255) :: fname
499  character(len= 6) :: csnapshot
500  character(len= 4) :: vname
501 !
502 !
503 !*********************************************************************************************************************************
504 !---->before writing snapshots, displacement, velocity and acceleration are interpolated at the snapshot_receiver x,y,z location
505 !*********************************************************************************************************************************
506 
507  do irec = 1,ig_nreceiver_snapshot
508 
509 !
510 !------->get GLL nodes displacement, velocity and acceleration xyz-values from quadrangle element which contains the snapshot receiver
511  call get_quad_gll_value(rg_gll_displacement,tg_receiver_snapshot_quad(irec)%gll,gll_dis)
512  call get_quad_gll_value(rg_gll_velocity ,tg_receiver_snapshot_quad(irec)%gll,gll_vel)
513  call get_quad_gll_value(rg_gll_acceleration,tg_receiver_snapshot_quad(irec)%gll,gll_acc)
514 
515 !
516 !------->interpolate displacement at the snapshot receiver
517  call quad_lagrange_interp(gll_dis,tg_receiver_snapshot_quad(irec)%lag,ux(irec),uy(irec),uz(irec))
518 !
519 !------->compute PGD in x, y and z directions
520  if ( abs(ux(irec)) > tg_receiver_snapshot_quad(irec)%pgd_x ) tg_receiver_snapshot_quad(irec)%pgd_x = abs(ux(irec))
521  if ( abs(uy(irec)) > tg_receiver_snapshot_quad(irec)%pgd_y ) tg_receiver_snapshot_quad(irec)%pgd_y = abs(uy(irec))
522  if ( abs(uz(irec)) > tg_receiver_snapshot_quad(irec)%pgd_z ) tg_receiver_snapshot_quad(irec)%pgd_z = abs(uz(irec))
523 !
524 !------->interpolate velocity at the snapshot receiver
525  call quad_lagrange_interp(gll_vel,tg_receiver_snapshot_quad(irec)%lag,vx(irec),vy(irec),vz(irec))
526 !
527 !------->compute PGV in x, y and z directions
528  if ( abs(vx(irec)) > tg_receiver_snapshot_quad(irec)%pgv_x ) tg_receiver_snapshot_quad(irec)%pgv_x = abs(vx(irec))
529  if ( abs(vy(irec)) > tg_receiver_snapshot_quad(irec)%pgv_y ) tg_receiver_snapshot_quad(irec)%pgv_y = abs(vy(irec))
530  if ( abs(vz(irec)) > tg_receiver_snapshot_quad(irec)%pgv_z ) tg_receiver_snapshot_quad(irec)%pgv_z = abs(vz(irec))
531  if ( sqrt(vx(irec)**2 + vy(irec)**2 + vz(irec)**2) > tg_receiver_snapshot_quad(irec)%pgv_xyz ) tg_receiver_snapshot_quad(irec)%pgv_xyz = sqrt(vx(irec)**2 + vy(irec)**2 + vz(irec)**2)
532 !
533 !------->interpolate acceleration at the snapshot receiver
534  call quad_lagrange_interp(gll_acc,tg_receiver_snapshot_quad(irec)%lag,ax(irec),ay(irec),az(irec))
535 !
536 !------->compute PGA in x, y and z directions
537  if ( abs(ax(irec)) > tg_receiver_snapshot_quad(irec)%pga_x ) tg_receiver_snapshot_quad(irec)%pga_x = abs(ax(irec))
538  if ( abs(ay(irec)) > tg_receiver_snapshot_quad(irec)%pga_y ) tg_receiver_snapshot_quad(irec)%pga_y = abs(ay(irec))
539  if ( abs(az(irec)) > tg_receiver_snapshot_quad(irec)%pga_z ) tg_receiver_snapshot_quad(irec)%pga_z = abs(az(irec))
540 
541  enddo
542 !
543 !
544 !*********************************************************************************************************************************
545 !---->write snapshot in gmt or VTK format
546 !*********************************************************************************************************************************
547  write(csnapshot,'(i6.6)') ig_idt
548 !
549 !
550 !---->snapshot displacement
551  if ( lg_snapshot_displacement .and. (mod(ig_idt-1,ig_snapshot_saving_incr) == 0) ) then
552 
553  if (lg_snapshot_gmt) then
554 
555  fname = trim(cg_prefix)//".snapshot.ux."//trim(csnapshot)//".grd"
556  call write_snapshot_gmt_native_format(fname,ux)
