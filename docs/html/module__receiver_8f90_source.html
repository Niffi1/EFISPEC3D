136 !***********************************************************************************************************************************************************************************
137  subroutine init_hexa_receiver()
138 !***********************************************************************************************************************************************************************************
139 
140  use mpi
141 
142  use mod_global_variables, only :&
143  cg_prefix&
144  ,tg_receiver_hexa&
145  ,ig_nreceiver_hexa&
146  ,ig_lst_unit&
147  ,ig_myrank&
148  ,ig_ncpu&
149  ,type_receiver_hexa&
150  ,ig_receiver_saving_incr&
151  ,lg_output_debug_file&
152  ,cg_myrank&
153  ,ig_hexa_receiver_unit&
154  ,get_newunit&
155  ,ig_ndt
156 
157  implicit none
158 
159  real , dimension(ig_ncpu) :: rldum1
160  real :: maxdmin
161  real :: rldmin
162 
163  integer, dimension(1) :: min_loc
164  integer, dimension(ig_ncpu) :: ilrcpu
165  integer :: ios
166  integer :: irec
167  integer :: i
168  integer :: j
169  integer :: ilnrec
170  integer, dimension(MPI_STATUS_SIZE) :: statut
171 
172  character(len=1) :: ctmp
173  character(len=6) :: crec
174 
175  type(type_receiver_hexa), allocatable, dimension(:) :: tlrinf
176 
177 !
178 !---->initialize total number of receivers in cpu myrank
179  ig_nreceiver_hexa = 0
180 !
181 !
182 !---->if file *.vor exists, then we read receiver x,y,z location
183  open(unit=100,file=trim(cg_prefix)//".vor",status='old',iostat=ios)
184 
185  if( (ios == 0) .and. (ig_receiver_saving_incr <= ig_ndt) ) then
186 
187  if (ig_myrank == 0) then
188  write(ig_lst_unit,'(a)') " "
189  write(ig_lst_unit,'(a)') "information about receivers in file "//trim(cg_prefix)//".vor"
190  endif
191 
192  do while (.true.)
193  read(unit=100,fmt=*,iostat=ios) ctmp
194  if (ios /= 0) exit
195  ig_nreceiver_hexa = ig_nreceiver_hexa + 1
196  enddo
197  rewind(100)
198 
199  allocate(tlrinf(ig_nreceiver_hexa))
200  ilnrec = 0
201 
202  do irec = 1,ig_nreceiver_hexa
203 
204 !
205 !---------->initialize tlrinf
206  tlrinf(irec)%x = 0.0
207  tlrinf(irec)%y = 0.0
208  tlrinf(irec)%z = 0.0
209  tlrinf(irec)%xi = 0.0
210  tlrinf(irec)%eta = 0.0
211  tlrinf(irec)%zeta = 0.0
212  tlrinf(irec)%dmin = 0.0
213  tlrinf(irec)%lag(:,:,:) = 0.0
214  tlrinf(irec)%gll(:,:,:) = 0
215  tlrinf(irec)%cpu = 0
216  tlrinf(irec)%iel = 0
217  tlrinf(irec)%kgll = 0
218  tlrinf(irec)%lgll = 0
219  tlrinf(irec)%mgll = 0
220  tlrinf(irec)%rglo = 0
221 
222  read(unit=100,fmt=*) tlrinf(irec)%x,tlrinf(irec)%y,tlrinf(irec)%z
223 !
224 !---------->find the closest element and gll point to the receiver irec
225  call search_closest_hexa_gll(tlrinf(irec)%x ,tlrinf(irec)%y ,tlrinf(irec)%z ,tlrinf(irec)%dmin&
226  ,tlrinf(irec)%kgll,tlrinf(irec)%lgll,tlrinf(irec)%mgll,tlrinf(irec)%iel )
227 !
228 !---------->check to which cpu the receiver belongs (i.e., cpu which has the smallest dmin)
229  call mpi_allgather(tlrinf(irec)%dmin,1,mpi_real,rldum1,1,mpi_real,mpi_comm_world,ios)
230  min_loc = minloc(rldum1(1:ig_ncpu))
231 
232  if (min_loc(1) == ig_myrank+1) then
233  tlrinf(irec)%cpu = ig_myrank + 1
234  ilnrec = ilnrec + 1
235  if (ig_myrank /= 0) call mpi_send(tlrinf(irec)%iel,1,mpi_integer,0,100,mpi_comm_world,ios)
236  else
237  tlrinf(irec)%cpu = min_loc(1)
238  endif
239 !
240 !---------->write some info about receivers in *.lst
241  if (ig_myrank == 0) then
242  if (min_loc(1) /= ig_myrank+1) call mpi_recv(tlrinf(irec)%iel,1,mpi_integer,min_loc(1)-1,100,mpi_comm_world,statut,ios)
243  write(ig_lst_unit,'(3(a,i8))') "receiver ",irec," computed by cpu ",min_loc(1)-1," hexa ",tlrinf(irec)%iel
244  endif
245 
246  enddo
247  close(100)
248 !
249 !------->transfer array tlrinf which knows all the receivers to array tg_receiver_hexa that needs only to know the receiver of cpu myrank
250  if (ilnrec > 0) then
251  allocate(tg_receiver_hexa(ilnrec))
252  j = 0
253  do irec = 1,ig_nreceiver_hexa
254  if (tlrinf(irec)%cpu == ig_myrank+1) then
255  j = j + 1
256  tg_receiver_hexa(j) = tlrinf(irec)
257  tg_receiver_hexa(j)%rglo = irec
258  endif
259  enddo
260  ig_nreceiver_hexa = ilnrec
261  else
262  ig_nreceiver_hexa = 0
263  endif
264  deallocate(tlrinf)
265 !
266 !------->cpu 0 gather the number of receiver (ig_nreceiver_hexa) of the other cpus
267  call mpi_gather(ig_nreceiver_hexa,1,mpi_integer,ilrcpu,1,mpi_integer,0,mpi_comm_world,ios)
268 !
269 !------->compute information for receivers that belongs to cpu myrank and send 'dmin' to cpu 0
270  do irec = 1,ig_nreceiver_hexa
271 
272  call compute_info_hexa_receiver(tg_receiver_hexa(irec))
273 
274  if (ig_myrank /= 0) then
275  call mpi_send(tg_receiver_hexa(irec)%dmin,1,mpi_real,0,101,mpi_comm_world,ios)
276  endif
277 
278  enddo
279 
280  if (lg_output_debug_file) then
281  open(unit=100,file="debug.cpu."//trim(cg_myrank)//".volume.receivers.info")
282  do irec = 1,ig_nreceiver_hexa
283  write(unit=100,fmt='(/,a,i10 )') "Receiver ",tg_receiver_hexa(irec)%rglo
284  write(unit=100,fmt='( a,i10 )') " in core ",tg_receiver_hexa(irec)%cpu-1
285  write(unit=100,fmt='( a,i10 )') " in hexa ",tg_receiver_hexa(irec)%iel
286  write(unit=100,fmt='( 3a )') " X "," Y "," Z "
287  write(unit=100,fmt='(3(e14.7,1x))') tg_receiver_hexa(irec)%x,tg_receiver_hexa(irec)%y,tg_receiver_hexa(irec)%z
288  write(unit=100,fmt='( 3a )') " XI "," ETA "," ZETA "
289  write(unit=100,fmt='(3(e14.7,1x))') tg_receiver_hexa(irec)%xi,tg_receiver_hexa(irec)%eta,tg_receiver_hexa(irec)%zeta
290  enddo
291  close(100)
292  endif
293 !
294 !------->cpu 0 gathers dmin of all cpus and write the maximum dmin in *.lst
295  if (ig_myrank == 0) then
296  maxdmin = 0.0
297  do i = 1,ig_ncpu
298  if (i == 1) then
299  do irec = 1,ig_nreceiver_hexa
300  maxdmin = max(maxdmin,tg_receiver_hexa(irec)%dmin)
301  enddo
302  else
303  do irec = 1,ilrcpu(i)
304  call mpi_recv(rldmin,1,mpi_real,i-1,101,mpi_comm_world,statut,ios)
305  maxdmin = max(maxdmin,rldmin)
306  enddo
307  endif
308  enddo
309  write(ig_lst_unit,'(a,e14.7)') "maximum localisation error of all receivers in volume = ",maxdmin
310  call flush(ig_lst_unit)
311  endif
312 !
313 !------->open receivers file once and for all
314  if ( ig_nreceiver_hexa > 0 ) then
315 
316  allocate(ig_hexa_receiver_unit(ig_nreceiver_hexa))
317 
318  do irec = 1,ig_nreceiver_hexa
319 
320  write(crec,'(i6.6)') tg_receiver_hexa(irec)%rglo
321 
322  open(unit = get_newunit(ig_hexa_receiver_unit(irec))&
323  ,file = trim(cg_prefix)//".vor."//trim(adjustl(crec))//".gpl"&
324  ,status = 'replace'&
325  ,action = 'write'&
326  ,access = 'sequential'&
327  ,form = 'unformatted'&
328  ,buffered = 'yes'&
329  ,recordtype = 'stream')
330 
331  enddo
332 
333  endif
334 
335  else
336  if (ig_myrank == 0) then
337  write(ig_lst_unit,'(/a)') "no volume receiver computed"
338  call flush(ig_lst_unit)
339  endif
340  endif
341 
342  return
343 !***********************************************************************************************************************************************************************************
344  end subroutine init_hexa_receiver

355  subroutine init_quad_receiver()
356 !***********************************************************************************************************************************************************************************
357 
358  use mpi
359 
360  use mod_global_variables, only :&
361  cg_prefix&
362  ,tg_receiver_quad&
363  ,ig_quadf_gll_glonum&
364  ,ig_nreceiver_quad&
365  ,ig_lst_unit&
366  ,ig_myrank&
367  ,ig_ncpu&
368  ,type_receiver_quad&
369  ,ig_receiver_saving_incr&
370  ,lg_output_debug_file&
371  ,cg_myrank&
372  ,ig_quad_receiver_unit&
373  ,get_newunit&
374  ,ig_ndt
375 
376  implicit none
377 
378  real , dimension(ig_ncpu) :: rldum1
379  real :: maxdmin
380  real :: rldmin
381 
382  integer, dimension(1) :: min_loc
383  integer, dimension(ig_ncpu) :: ilrcpu
384  integer :: ios
385  integer :: irec
386  integer :: i
387  integer :: j
388  integer :: ilnrec
389  integer, dimension(MPI_STATUS_SIZE) :: statut
390

392  character(len=6) :: crec
393 
394  type(type_receiver_quad), allocatable, dimension(:) :: tlrinf
395 
396 !
397 !---->initialize total number of receivers in cpu myrank
398  ig_nreceiver_quad = 0
399 !
400 !
401 !---->if file *receiver exists, then we read receiver x,y,z location
402  open(unit=100,file=trim(cg_prefix)//".fsr",status='old',iostat=ios)
403 
404  if( (ios == 0) .and. (ig_receiver_saving_incr <= ig_ndt) ) then
405 
406  if (ig_myrank == 0) then
407  write(ig_lst_unit,'(a)') " "
408  write(ig_lst_unit,'(a)') "information about receivers in file "//trim(cg_prefix)//".fsr"
409  endif
410 
411  do while (.true.)
412  read(unit=100,fmt=*,iostat=ios) ctmp
413  if (ios /= 0) exit
414  ig_nreceiver_quad = ig_nreceiver_quad + 1
415  enddo
416  rewind(100)
417 
418  allocate(tlrinf(ig_nreceiver_quad))
419  ilnrec = 0
420 
421  do irec = 1,ig_nreceiver_quad
422 
423 !
424 !---------->initialize tlrinf
425  tlrinf(irec)%x = 0.0
426  tlrinf(irec)%y = 0.0
427  tlrinf(irec)%z = 0.0
428  tlrinf(irec)%xi = 0.0
429  tlrinf(irec)%eta = 0.0
430  tlrinf(irec)%dmin = 0.0
431  tlrinf(irec)%lag(:,:) = 0.0
432  tlrinf(irec)%gll(:,:) = 0
433  tlrinf(irec)%cpu = 0
434  tlrinf(irec)%iel = 0
435  tlrinf(irec)%lgll = 0
436  tlrinf(irec)%mgll = 0
437  tlrinf(irec)%rglo = 0
438 
439  read(unit=100,fmt=*) tlrinf(irec)%x,tlrinf(irec)%y
440 !
441 !---------->find the closest element and gll point to the receiver irec
442  call search_closest_quad_gll(tlrinf(irec)%x,tlrinf(irec)%y,ig_quadf_gll_glonum,tlrinf(irec)%dmin,tlrinf(irec)%lgll,tlrinf(irec)%mgll,tlrinf(irec)%iel)
443 !
444 !---------->check to which cpu the receiver belongs (i.e., cpu which has the smallest dmin)
445  call mpi_allgather(tlrinf(irec)%dmin,1,mpi_real,rldum1,1,mpi_real,mpi_comm_world,ios)
446  min_loc = minloc(rldum1(1:ig_ncpu))
447 
448  if (min_loc(1) == ig_myrank+1) then
449  tlrinf(irec)%cpu = ig_myrank + 1
450  ilnrec = ilnrec + 1
451  if (ig_myrank /= 0) call mpi_send(tlrinf(irec)%iel,1,mpi_integer,0,110,mpi_comm_world,ios)
452  else
453  tlrinf(irec)%cpu = min_loc(1)
454  endif
455 !
456 !---------->write some info about receivers in *.lst
457  if (ig_myrank == 0) then
458  if (min_loc(1) /= ig_myrank+1) call mpi_recv(tlrinf(irec)%iel,1,mpi_integer,min_loc(1)-1,110,mpi_comm_world,statut,ios)
459  write(ig_lst_unit,'(3(a,i8))') "receiver ",irec," computed by cpu ",min_loc(1)-1," quad ",tlrinf(irec)%iel
460  endif
461 
462  enddo
463  close(100)
464 !
465 !------->transfer array tlrinf which knows all the receivers to array tg_receiver_quad that needs only to know the receiver of cpu myrank
466  if (ilnrec > 0) then
467  allocate(tg_receiver_quad(ilnrec))
468  j = 0
469  do irec = 1,ig_nreceiver_quad
470  if (tlrinf(irec)%cpu == ig_myrank+1) then
471  j = j + 1
472  tg_receiver_quad(j) = tlrinf(irec)
473  tg_receiver_quad(j)%rglo = irec
474  endif
475  enddo
476  ig_nreceiver_quad = ilnrec
477  else
478  ig_nreceiver_quad = 0
479  endif
480  deallocate(tlrinf)
481 !
482 !------->cpu 0 gather the number of receiver (ig_nreceiver_quad) of the other cpus
483  call mpi_gather(ig_nreceiver_quad,1,mpi_integer,ilrcpu,1,mpi_integer,0,mpi_comm_world,ios)
484 !
485 !------->compute information for receivers that belongs to cpu myrank and send 'dmin' to cpu 0
486  do irec = 1,ig_nreceiver_quad
487 
488  call compute_info_quad_receiver(tg_receiver_quad(irec))
489 
490  if (ig_myrank /= 0) then
491  call mpi_send(tg_receiver_quad(irec)%dmin,1,mpi_real,0,111,mpi_comm_world,ios)
492  endif
493 
494  enddo
495 
496  if (lg_output_debug_file) then
497  open(unit=100,file="debug.cpu."//trim(cg_myrank)//".freesurface.receivers.info")
498  do irec = 1,ig_nreceiver_quad
499  write(unit=100,fmt='(/,a,i10 )') "Receiver ",tg_receiver_quad(irec)%rglo
500  write(unit=100,fmt='( a,i10 )') " in core ",tg_receiver_quad(irec)%cpu-1
501  write(unit=100,fmt='( a,i10 )') " in quad ",tg_receiver_quad(irec)%iel
502  write(unit=100,fmt='( 2a )') " X "," Y "
503  write(unit=100,fmt='(2(e14.7,1x))') tg_receiver_quad(irec)%x,tg_receiver_quad(irec)%y
504  write(unit=100,fmt='( 2a )') " XI "," ETA "
505  write(unit=100,fmt='(2(e14.7,1x))') tg_receiver_quad(irec)%xi,tg_receiver_quad(irec)%eta
506  enddo
507  close(100)
508  endif
509 !
510 !------->cpu 0 gathers dmin of all cpus and write the maximum dmin in *.lst
511  if (ig_myrank == 0) then
512  maxdmin = 0.0
513  do i = 1,ig_ncpu
514  if (i == 1) then
515  do irec = 1,ig_nreceiver_quad
516  maxdmin = max(maxdmin,tg_receiver_quad(irec)%dmin)
517  enddo
518  else
519  do irec = 1,ilrcpu(i)
520  call mpi_recv(rldmin,1,mpi_real,i-1,111,mpi_comm_world,statut,ios)
521  maxdmin = max(maxdmin,rldmin)
522  enddo
523  endif
524  enddo
525  write(ig_lst_unit,'(a,e14.7)') "maximum localisation error of all receivers in free surface = ",maxdmin
526  call flush(ig_lst_unit)
527  endif
528 !
529 !------->open receivers file once and for all
530  if ( ig_nreceiver_quad > 0 ) then
531 
532  allocate(ig_quad_receiver_unit(ig_nreceiver_quad))
533 
534  do irec = 1,ig_nreceiver_quad
535 
536  write(crec,'(i6.6)') tg_receiver_quad(irec)%rglo
537 
538  open(unit = get_newunit(ig_quad_receiver_unit(irec))&
539  ,file = trim(cg_prefix)//".fsr."//trim(adjustl(crec))//".gpl"&
540  ,status = 'replace'&
541  ,action = 'write'&
542  ,access = 'sequential'&
543  ,form = 'unformatted'&
544  ,buffered = 'yes'&
545  ,recordtype = 'stream')
546 
547  enddo
548 
549  endif
550 
551  else
552  if (ig_myrank == 0) then
553  write(ig_lst_unit,'(/a)') "no free surface receiver computed"
554  call flush(ig_lst_unit)
555  endif
556  endif
557 
558  return
559 !***********************************************************************************************************************************************************************************
560  end subroutine init_quad_receiver
