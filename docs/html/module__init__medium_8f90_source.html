165  ,rg_hexa_gll_ksixx&
166  ,rg_hexa_gll_ksiyy&

190  integer :: ios

192  integer :: imat
193  integer :: imem_var

205  !
206  !************************************************************************************************
207  !initialize memory
208  !************************************************************************************************
209 
210  ios = init_array_real(rg_hexa_gll_rho,ig_nhexa,ig_ngll,ig_ngll,ig_ngll,"rg_hexa_gll_rho") !deallocated in subroutine init_mass_matrix.f90
211 
212  ios = init_array_real(rg_hexa_gll_rhovs2,ig_nhexa,ig_ngll,ig_ngll,ig_ngll,"rg_hexa_gll_rhovs2")
213 
214  ios = init_array_real(rg_hexa_gll_rhovp2,ig_nhexa,ig_ngll,ig_ngll,ig_ngll,"rg_hexa_gll_rhovp2")
215 
216 
217  if (lg_visco) then
218 
219  ios = init_array_real(rg_hexa_gll_ksixx,ig_nhexa,ig_ngll,ig_ngll,ig_ngll,ig_nrelax,"rg_hexa_gll_ksixx")
220 
221  ios = init_array_real(rg_hexa_gll_ksiyy,ig_nhexa,ig_ngll,ig_ngll,ig_ngll,ig_nrelax,"rg_hexa_gll_ksiyy")
222 
223  ios = init_array_real(rg_hexa_gll_ksizz,ig_nhexa,ig_ngll,ig_ngll,ig_ngll,ig_nrelax,"rg_hexa_gll_ksizz")
224 
225  ios = init_array_real(rg_hexa_gll_ksixy,ig_nhexa,ig_ngll,ig_ngll,ig_ngll,ig_nrelax,"rg_hexa_gll_ksixy")
226 
227  ios = init_array_real(rg_hexa_gll_ksixz,ig_nhexa,ig_ngll,ig_ngll,ig_ngll,ig_nrelax,"rg_hexa_gll_ksixz")
228 
229  ios = init_array_real(rg_hexa_gll_ksiyz,ig_nhexa,ig_ngll,ig_ngll,ig_ngll,ig_nrelax,"rg_hexa_gll_ksiyz")
230 
231  ios = init_array_real(rg_hexa_gll_wkqs,ig_nhexa,ig_ngll,ig_ngll,ig_ngll,ig_nrelax,"rg_hexa_gll_wkqs")
232 
233  ios = init_array_real(rg_hexa_gll_wkqp,ig_nhexa,ig_ngll,ig_ngll,ig_ngll,ig_nrelax,"rg_hexa_gll_wkqp")
234 
235 !
236 !------->precompute exponential used for computing internal forces
237  do imem_var = 1,ig_nrelax
238  rg_mem_var_exp(imem_var) = exp(-rg_dt/rg_relax_coeff(imem_var,1))
239  enddo
240 
241  endif
242 
243 
244  !
245  !
246  !***********************************************************************************************************
247  !->if a geomodeler file *.vox or *.mat exists, then the medium (ig_hexa_material_number) is read from such files
248  !***********************************************************************************************************
249  is_file_exist = .false.
250 
251  inquire(file=trim(cg_prefix)//".cpu."//trim(cg_myrank)//".mat", exist=is_file_exist)
252  if (is_file_exist) then
253  if (ig_myrank == 0) then
254  write(ig_lst_unit,'(a)') " --> medium generated from geomodeler's material (file *.mat)"
255  endif
256  call init_medium_from_mat_file(ig_hexa_material_number)
257  endif
258 
259  if ( .not.(is_file_exist) .and. (ig_myrank == 0) ) then
260  write(ig_lst_unit,'(a)') " --> medium generated from cubit block's information"
261  endif
262 
263 
264  !
265  !
266  !************************************************************************************************
267  !->check if the medium (ig_hexa_material_number) is entirely defined
268  !************************************************************************************************
269  do iel = 1,ig_nhexa
270 
271  imat = ig_hexa_material_number(iel)
272 
273  if ( (imat <= 0) .or. (imat > ig_nmaterial) ) then
274  write(info,'(a)') "error in subroutine init_hexa_medium: undefined material for hexa"
275  call error_stop(info)
276  endif
277 
278  enddo
279 
280 
281  !
282  !
283  !************************************************************************************************
284  !Compute elastic and viscoelastic parameter pour media_type = 0
285  !************************************************************************************************
286  do imat = 1,ig_nmaterial
287 !
288 !
289 !------->elastic
290  tg_elastic_material(imat)%lam2 = tg_elastic_material(imat)%dens*tg_elastic_material(imat)%svel**2
291  tg_elastic_material(imat)%lam1 = tg_elastic_material(imat)%dens*tg_elastic_material(imat)%pvel**2 - 2.0*tg_elastic_material(imat)%lam2
292  tg_elastic_material(imat)%pois = (tg_elastic_material(imat)%pvel**2 - 2.0*tg_elastic_material(imat)%svel**2)/(2.0*(tg_elastic_material(imat)%pvel**2-tg_elastic_material(imat)%svel**2))
293  tg_elastic_material(imat)%youn = 2.0*tg_elastic_material(imat)%lam2*(1.0+tg_elastic_material(imat)%pois)
294  tg_elastic_material(imat)%bulk = tg_elastic_material(imat)%lam1+2.0/3.0*tg_elastic_material(imat)%lam2
295  tg_elastic_material(imat)%rvel = (0.862+1.14*tg_elastic_material(imat)%pois)/(1.0+tg_elastic_material(imat)%pois)*tg_elastic_material(imat)%svel
296  tg_elastic_material(imat)%pwmo = tg_elastic_material(imat)%lam1 + 2.0*tg_elastic_material(imat)%lam2
297 
298  if (ig_myrank == 0) then
299  if (ig_material_type(imat) == 1) write(ig_lst_unit,'(" ",/,a,i0,a)') "material ",imat," is elastic"
300  if (ig_material_type(imat) == 2) write(ig_lst_unit,'(" ",/,a,i0,a)') "material ",imat," is viscoelactic"
301  write(ig_lst_unit,'(a,f15.8 ,a)') "s-wave velocity = ",tg_elastic_material(imat)%svel," m/s"
302  write(ig_lst_unit,'(a,f15.8 ,a)') "p-wave velocity = ",tg_elastic_material(imat)%pvel," m/s"
303  write(ig_lst_unit,'(a,f15.8 ,a)') "poisson ratio = ",tg_elastic_material(imat)%pois," "
304  write(ig_lst_unit,'(a,f15.8 ,a)') "density = ",tg_elastic_material(imat)%dens," kg/m3"
305  write(ig_lst_unit,'(a,e15.7e3,a)') "lambda (first lame coef.) = ",tg_elastic_material(imat)%lam1," Pa"
306  write(ig_lst_unit,'(a,e15.7e3,a)') "s-wave modulus = ",tg_elastic_material(imat)%lam2," Pa"
307  write(ig_lst_unit,'(a,e15.7e3,a)') "p-wave modulus = ",tg_elastic_material(imat)%pwmo," Pa"
308  write(ig_lst_unit,'(a,e15.7e3,a)') "bulk-modulus = ",tg_elastic_material(imat)%bulk," Pa"
309  endif
310 !
311 !
312 !------->viscoelastic
313  if (lg_visco) then
314 
315  allocate(tg_viscoelastic_material(imat)%wkqs(ig_nrelax),tg_viscoelastic_material(imat)%wkqp(ig_nrelax))
316 
317  dtmpqs = (3.071+1.433*tg_viscoelastic_material(imat)%qs**(-1.158)*log(tg_viscoelastic_material(imat)%qs/5.0))/(1.0+0.415*tg_viscoelastic_material(imat)%qs)
318  dtmpqp = (3.071+1.433*tg_viscoelastic_material(imat)%qp**(-1.158)*log(tg_viscoelastic_material(imat)%qp/5.0))/(1.0+0.415*tg_viscoelastic_material(imat)%qp)
319  do k = 1,ig_nrelax
320  tg_viscoelastic_material(imat)%wkqs(k) = dtmpqs*(dtmpqs*rg_relax_coeff(k,2)+rg_relax_coeff(k,3))
321  tg_viscoelastic_material(imat)%wkqp(k) = dtmpqp*(dtmpqp*rg_relax_coeff(k,2)+rg_relax_coeff(k,3))
322  enddo
323 
324  ctmpqs = cmplx(0.0,0.0)
325  ctmpqp = cmplx(0.0,0.0)
326  do k = 1,ig_nrelax
327  ctmpqs = ctmpqs + cmplx(tg_viscoelastic_material(imat)%wkqs(k),0.0)/(1.0 + cmplx(0.0,1.0)*2.0*rg_pi*tg_viscoelastic_material(imat)%freq*rg_relax_coeff(k,1))
328  ctmpqp = ctmpqp + cmplx(tg_viscoelastic_material(imat)%wkqp(k),0.0)/(1.0 + cmplx(0.0,1.0)*2.0*rg_pi*tg_viscoelastic_material(imat)%freq*rg_relax_coeff(k,1))
329  enddo
330 
331  tg_viscoelastic_material(imat)%uswm = tg_elastic_material(imat)%lam2/abs(1.0-ctmpqs)
332  tg_viscoelastic_material(imat)%upwm = tg_elastic_material(imat)%pwmo/abs(1.0-ctmpqp)
333  if (ig_myrank == 0) then
334  write(ig_lst_unit,'(a,f15.8 ,a)') "frequency for viscosity = ",tg_viscoelastic_material(imat)%freq," hz"
335  write(ig_lst_unit,'(a,f15.8 ,a)') "s-wave quality factor = ",tg_viscoelastic_material(imat)%qs ," "
336  write(ig_lst_unit,'(a,f15.8 ,a)') "p-wave quality factor = ",tg_viscoelastic_material(imat)%qp ," "
337  write(ig_lst_unit,'(a,e15.7e3,a)') "unrelaxed s-wave modulus = ",tg_viscoelastic_material(imat)%uswm," pa"
338  write(ig_lst_unit,'(a,e15.7e3,a)') "unrelaxed p-wave modulus = ",tg_viscoelastic_material(imat)%upwm," pa"
339  endif
340 
341  endif !endif of material type viscoelastic
342 
343  enddo !enddo of loop on material
344 
345 
346  !
347  !
348  !************************************************************************************************
349  !Distribute material properties to gll nodes
350  !************************************************************************************************
351 !
352 !
353 !------->elastic
354  if (.not.lg_visco) then
355  do iel = 1,ig_nhexa
356  do k = 1,ig_ngll !zeta
357  do l = 1,ig_ngll !eta
358  do m = 1,ig_ngll !xi
359 
360  imat = ig_hexa_material_number(iel)
361  rg_hexa_gll_rho(m,l,k,iel) = tg_elastic_material(imat)%dens
362  rg_hexa_gll_rhovs2(m,l,k,iel) = tg_elastic_material(imat)%dens*tg_elastic_material(imat)%svel**2
363  rg_hexa_gll_rhovp2(m,l,k,iel) = tg_elastic_material(imat)%dens*tg_elastic_material(imat)%pvel**2
364 
365  enddo
366  enddo
367  enddo
368  enddo
369 !
370 !
371 !------->viscoelastic
372  else
373  do iel = 1,ig_nhexa
374  do k = 1,ig_ngll !zeta
375  do l = 1,ig_ngll !eta
376  do m = 1,ig_ngll !xi
377 
378  imat = ig_hexa_material_number(iel)
379  rg_hexa_gll_rho(m,l,k,iel) = tg_elastic_material(imat)%dens
380  rg_hexa_gll_rhovs2(m,l,k,iel) = tg_viscoelastic_material(imat)%uswm
381  rg_hexa_gll_rhovp2(m,l,k,iel) = tg_viscoelastic_material(imat)%upwm
382 
383  do n = 1,ig_nrelax
384  rg_hexa_gll_wkqs(n,m,l,k,iel) = tg_viscoelastic_material(imat)%wkqs(n)
385  rg_hexa_gll_wkqp(n,m,l,k,iel) = tg_viscoelastic_material(imat)%wkqp(n)
386  enddo
387 
388  enddo
389  enddo
390  enddo
391  enddo
392  endif
393 
394  !
395  !
396  !************************************************************************************************
397  !write material in VTK xml format
398  !************************************************************************************************
399 
400  if (lg_output_medium_vtk) call write_medium_vtk_cell_xml()
401 
402  return
403 !***********************************************************************************************************************************************************************************
404  end subroutine init_hexa_medium

414  subroutine init_quadp_medium()
415 !***********************************************************************************************************************************************************************************
416  use mpi
417  use mod_global_variables, only :&
418  ig_ngll&
419  ,ig_myrank&
420  ,ig_nquad_parax&
421  ,ig_quadp_neighbor_hexa&
422  ,ig_quadp_neighbor_hexaface&
423  ,rg_quadp_gll_rhovs&
424  ,rg_quadp_gll_rhovp&
425  ,rg_hexa_gll_rho&
426  ,rg_hexa_gll_rhovs2&
427  ,rg_hexa_gll_rhovp2&
428  ,error_stop&
429  ,lg_boundary_absorption
430 
431  use mod_init_memory
432 
433  implicit none
434 
435  integer :: ios
436  integer :: i
437  integer :: j
438  integer :: k
439  integer :: iquad
440  integer :: ihexa
441  integer :: iface
442 
443  character(len=255) :: info
444 
445  !
446  !See numbering convention in subroutine propagate_gll_nodes_quad of module_mesh_gll_num.f90
447  !
448 
449  if (ig_nquad_parax > 0 .and. lg_boundary_absorption) then
450 
451  ios = init_array_real(rg_quadp_gll_rhovs,ig_nquad_parax,ig_ngll,ig_ngll,"rg_quadp_gll_rhovs")
452 
453  ios = init_array_real(rg_quadp_gll_rhovp,ig_nquad_parax,ig_ngll,ig_ngll,"rg_quadp_gll_rhovp")
454 
455  do iquad = 1,ig_nquad_parax
456 
457  ihexa = ig_quadp_neighbor_hexa(iquad)
458  iface = ig_quadp_neighbor_hexaface(iquad)
459 
460  select case(iface)
461  case(1)
462  do j = 1,ig_ngll
463  do i = 1,ig_ngll
464  rg_quadp_gll_rhovs(i,j,iquad) = rg_hexa_gll_rho(i,j,1,ihexa)*sqrt(rg_hexa_gll_rhovs2(i,j,1,ihexa)/rg_hexa_gll_rho(i,j,1,ihexa))
465  rg_quadp_gll_rhovp(i,j,iquad) = rg_hexa_gll_rho(i,j,1,ihexa)*sqrt(rg_hexa_gll_rhovp2(i,j,1,ihexa)/rg_hexa_gll_rho(i,j,1,ihexa))
466  enddo
467  enddo
468 
469  case(2)
470  do k = 1,ig_ngll
471  do i = 1,ig_ngll
472  rg_quadp_gll_rhovs(k,i,iquad) = rg_hexa_gll_rho(i,1,k,ihexa)*sqrt(rg_hexa_gll_rhovs2(i,1,k,ihexa)/rg_hexa_gll_rho(i,1,k,ihexa))
473  rg_quadp_gll_rhovp(k,i,iquad) = rg_hexa_gll_rho(i,1,k,ihexa)*sqrt(rg_hexa_gll_rhovp2(i,1,k,ihexa)/rg_hexa_gll_rho(i,1,k,ihexa))
474  enddo
475  enddo
476 
477  case(3)
478  do k = 1,ig_ngll
479  do j = 1,ig_ngll
480  rg_quadp_gll_rhovs(k,j,iquad) = rg_hexa_gll_rho(ig_ngll,j,k,ihexa)*sqrt(rg_hexa_gll_rhovs2(ig_ngll,j,k,ihexa)/rg_hexa_gll_rho(ig_ngll,j,k,ihexa))
481  rg_quadp_gll_rhovp(k,j,iquad) = rg_hexa_gll_rho(ig_ngll,j,k,ihexa)*sqrt(rg_hexa_gll_rhovp2(ig_ngll,j,k,ihexa)/rg_hexa_gll_rho(ig_ngll,j,k,ihexa))
482  enddo
483  enddo
484 
485  case(4)
486  do k = 1,ig_ngll
487  do i = 1,ig_ngll
