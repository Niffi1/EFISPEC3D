523 !***********************************************************************************************************************************************************************************
524  subroutine init_gll_nodes()
525 !***********************************************************************************************************************************************************************************
526 
527  use mpi
528 
529  use mod_global_variables, only : &
530  cg_prefix&
531  ,rg_gll_abscissa_dist&
532  ,rg_gll_abscissa&
533  ,rg_gll_weight&
534  ,rg_gll_lagrange_deriv&
535  ,ig_ngll&
536  ,ig_lst_unit&
537  ,rg_pi&
538  ,ig_myrank&
539  ,ig_lagrange_order
540 
541  use mod_lagrange , only : lagrad
542 
543  implicit none
544 
545  real, parameter :: eps = 1.0e-7
546  real, allocatable ,dimension(:) :: xold
547  real, dimension(IG_NGLL,IG_NGLL) :: vandermonde_matrix
548  integer :: ngll
549  integer :: i
550  integer :: j
551  integer :: k
552  integer :: n
553 
554  !
555  !
556  !higher integration order among different group is taken as the integration order
557  n = ig_lagrange_order
558 
559 
560  !
561  !
562  !ngll is the number of integration node in 1d along the segment [-1;1]
563  ngll = n + 1
564 
565 
566  !
567  !
568  !*******************************
569  !find gll abscissa and weight
570  !*******************************
571  !
572  !use the chebyshev-gauss-lobatto nodes as the first guess
573  do i = 1,ngll
574  rg_gll_abscissa(i) = cos(rg_pi*real(i-1)/real(n))
575  enddo
576  !
577  !initialize the legendre vandermonde matrix
578  vandermonde_matrix(:,:) = 0.0
579  !
580  !compute p_{n} using the recursion relation. compute its first and second derivatives and update rg_gll_abscissa using the newton-raphson method
581  allocate(xold(ngll))
582  xold(:) = 2.0
583  do while (maxval(abs(rg_gll_abscissa-xold)) > eps)
584 
585  xold(:) = rg_gll_abscissa(:)
586  vandermonde_matrix(:,1) = 1.0
587  vandermonde_matrix(:,2) = rg_gll_abscissa(:)
588 
589  do k=2,n
590  vandermonde_matrix(:,k+1) = ( (2.0*real(k)-1.0)*rg_gll_abscissa(:)*vandermonde_matrix(:,k)-(real(k)-1.0)*vandermonde_matrix(:,k-1) )/real(k)
591  enddo
592 
593  rg_gll_abscissa(:) = xold(:)-( rg_gll_abscissa(:)*vandermonde_matrix(:,ngll)-vandermonde_matrix(:,n) )/( ngll*vandermonde_matrix(:,ngll) )
594 
595  enddo
596  !
597  !compute corresponding weight
598  rg_gll_weight(:) = 2.0/(n*ngll*vandermonde_matrix(:,ngll)**2)
599 
600  !
601  !
602  !*******************************************************************************
603  !write results rg_gll_abscissa,rg_gll_weight in file *.lst
604  !*******************************************************************************
605  if (ig_myrank == 0) then
606  write(ig_lst_unit,'(" ",/,a)') " --> abscissa of GLLs - weight of GLLs"
607  do i = 1,ngll
608  write(ig_lst_unit,fmt='(2(7X,f10.7))') rg_gll_abscissa(i),rg_gll_weight(i)
609  enddo
610  endif
611 
612  !
613  !
614  !*******************************************************************************
615  !compute subtraction between gll nodes (used in subroutines lagrap and lagrad)
616  !*******************************************************************************
617  do i = 1,ig_ngll
618  do j = 1,ig_ngll
619  rg_gll_abscissa_dist(j,i) = 0.0
620  if (i /= j) rg_gll_abscissa_dist(j,i) =1.0/(rg_gll_abscissa(i) - rg_gll_abscissa(j))
621  enddo
622  enddo
623 
624  !
625  !
626  !***************************************************
627  !compute derivative of lagrange poly at GLL nodes
628  !***************************************************
629  !
630  !notation convention: rg_gll_lagrange_deriv(j,i) = h'_j(xgll_i)
631  do i = 1,ig_ngll
632  do j = 1,ig_ngll
633  rg_gll_lagrange_deriv(j,i) = lagrad(j,rg_gll_abscissa(i),ig_ngll)
634  enddo
635  enddo
636 
637  return
638 
639 !***********************************************************************************************************************************************************************************
640  end subroutine init_gll_nodes
